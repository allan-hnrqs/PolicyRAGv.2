\begin{frame}{Problem}
  \begin{itemize}
    \item Procurement-policy QA is a high-exactness RAG problem.
    \item The system must retrieve relevant evidence and avoid inventing missing identifiers, forms, contacts, or workflow steps.
    \item Evaluation has to check both coverage and faithfulness.
  \end{itemize}
\end{frame}

\begin{frame}{Pipeline Overview}
  \begin{enumerate}
    \item Collect source pages
    \item Normalize and enrich documents
    \item Chunk them
    \item Build the hybrid retrieval index
    \item Retrieve and pack evidence
    \item Generate the answer
    \item Evaluate with multiple measurement lanes
  \end{enumerate}
  \vspace{0.5em}
  Current system: Buyer’s Guide-first hybrid RAG with query decomposition, strict
  evaluation, and a narrow exactness-specific post-draft correction path.
\end{frame}

\begin{frame}{Key Terms}
  \begin{itemize}
    \item \textbf{Canonical parity:} the frozen benchmark family used for tuning and promotion checks.
    \item \textbf{\texttt{parity19\_dev}:} the 9-case development split used during tuning.
    \item \textbf{\texttt{parity19\_holdout}:} the 10-case protected split used for promotion checks.
    \item \textbf{p95 / p99:} the 95th / 99th percentile. Example: p99 chunk size means 99\% of chunks are at or below that length.
    \item \textbf{Sliding-window chunking:} fixed-size text windows, usually with overlap, instead of structure-aware paragraph/section blocks.
    \item \textbf{Intervention-only evaluation:} baseline stays unchanged unless the conditional method actually fires.
  \end{itemize}
\end{frame}

\begin{frame}{Source Design}

    \item Buyer’s Guide is the primary retrieval surface.

\begin{frame}{Worked Example: Query Decomposition Helped Retrieval}
  \small
  \textbf{Case:} \texttt{HR\_008}

  \vspace{0.4em}
  \textbf{Prompt:}
  \begin{quote}
    What is the actual contractual difference between a standing offer and a
    supply arrangement? When does Canada become legally bound under each one?
  \end{quote}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Baseline retrieval}
      \begin{itemize}
        \item Top evidence was dominated by \emph{Using a standing offer}
        \item Example chunk:
      \end{itemize}
      \scriptsize
      ``A standing offer is not a contract. A contract is established with each
      call-up for goods or services made against the standing offer.''
      \normalsize

      \vspace{0.3em}
      \textbf{Result:} recall 0.3333
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{With query decomposition}
      \begin{itemize}
        \item Added targeted retrieval queries about standing offers vs supply arrangements
        \item Newly packed chunk:
      \end{itemize}
      \scriptsize
      ``There is no contractual obligation between Canada and the offeror to buy
      goods or services at the supply arrangement stage. There is a contractual
      obligation with the award of each contract.''
      \normalsize

      \vspace{0.3em}
      \textbf{Result:} recall 1.0
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Worked Example: Exactness Path Prevented an Unsupported Answer}
  \small
  \textbf{Case:} \texttt{HR\_037}

  \vspace{0.4em}
  \textbf{Prompt:}
  \begin{quote}
    What exact form number or file name do I need for the specimen signature
    card template in the GC network folder when approvals go beyond delegated
    authority?
  \end{quote}

  \textbf{Retrieved evidence excerpt:}
  \scriptsize
  ``The Delegation of Authorities instrument and Schedule 3 specimen signature
  card templates for the Acquisitions Program can be found in the Delegation of
  Authorities ... folder.''

  \vspace{0.3em}
  ``PSPC supervisors ... complete and sign a Schedule 3 Specimen Signature Card
  Template ... This form documents the scope of the incumbent's delegated
  approval and signing authority.''
  \normalsize

  \vspace{0.1em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Baseline answer}
      \begin{itemize}
        \item Gave a specific file/form name as if it were established
        \item Result:
      \end{itemize}
      \scriptsize
      ``The exact form number or file name ... is the Schedule 3 Specimen
      Signature Card Template.''
      \normalsize

      \vspace{0.3em}
      Recall 0.3333, forbidden 1
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Exactness path answer}
      \begin{itemize}
        \item Explicitly said the evidence does not provide the exact file name
        \item Result:
      \end{itemize}
      \scriptsize
      ``The exact form number or file name ... is not provided in the evidence.''
      \normalsize

      \vspace{0.3em}
      Recall 0.6667, forbidden 0
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Worked Example: Why Broad Mode-Aware Answering Was Rejected}
  \small
  \textbf{Case:} \texttt{HR\_001}

  \vspace{0.4em}
  \textbf{Prompt:}
  \begin{quote}
    We received a bid after the deadline and the supplier says they sent it on
    time. What proof counts, and what are we supposed to do with the offer?
  \end{quote}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Query-decomposition baseline}
      \begin{itemize}
        \item Covered acceptable proof
        \item Covered handling of late offers
        \item Covered offeror responsibility and return handling
      \end{itemize}

      \vspace{0.3em}
      \textbf{Result:} recall 1.0
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Mode-aware branch}
      \begin{itemize}
        \item Still covered acceptable proof
        \item But dropped part of the handling logic
        \item Judge notes: missing support for offeror responsibility and return based on delivery method
      \end{itemize}

      \vspace{0.3em}
      \textbf{Result:} recall 0.5
    \end{column}
  \end{columns}

  \vspace{0.5em}
  This is why rebuilt-39 gains alone were not enough for promotion.
\end{frame}
